macro(sourcemeta_one_test_cli_common program name)
  if(UNIX)
    add_test(NAME one.${program}.common.${name} COMMAND
      "${CMAKE_CURRENT_SOURCE_DIR}/${program}/common/${name}.sh"
      "$<TARGET_FILE:sourcemeta_one_${program}>"
      "${ONE_VERSION}"
      WORKING_DIRECTORY "${PROJECT_SOURCE_DIR}")
    set_tests_properties(one.${program}.common.${name} PROPERTIES
      ENVIRONMENT "ONE_PREFIX=${ONE_PREFIX}")
  endif()
endmacro()
